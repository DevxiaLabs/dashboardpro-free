<project_specification>
  <project_name>DashBoard Pro</project_name>

  <overview>
    A premium, modern and elegant dashboard template designed for sale on ThemeForest. It features public authentication pages (login, register, forgot password) and a private dashboard area with custom UI components, data visualizations, and multiple functional pages. Built with Next.js and Tailwind CSS, all data is mocked — there is no backend. The template targets developers and businesses looking for a production-ready admin dashboard starting point.
  </overview>

  <technology_stack>
    <frontend>
      <framework>Next.js 14+ (App Router)</framework>
      <styling>Tailwind CSS 3.4+</styling>
      <language>TypeScript</language>
      <charts>Recharts</charts>
      <icons>Lucide React</icons>
      <drag_and_drop>@dnd-kit/core + @dnd-kit/sortable</drag_and_drop>
      <date_handling>date-fns</date_handling>
      <typography>Inter (Google Fonts)</typography>
      <animations>Framer Motion (subtle transitions)</animations>
    </frontend>
    <backend>
      <runtime>None — stateless template with mock data</runtime>
      <database>none - stateless application</database>
    </backend>
    <communication>
      <api>No real API — all data is provided via local mock data files and React context/state</api>
    </communication>
  </technology_stack>

  <prerequisites>
    <environment_setup>
      - Node.js 18+ and npm/pnpm
      - Next.js project initialized with App Router
      - Tailwind CSS configured
      - TypeScript configured
    </environment_setup>
  </prerequisites>

  <feature_count>128</feature_count>

  <stateless_template_note>
    This is a frontend-only template with NO backend, NO database, and NO real API.
    All data is mocked using local JSON/TS files and React state/context.
    Authentication is simulated (no real auth provider).
    Infrastructure database features (indices 0-4) are replaced with template-specific foundation features.
  </stateless_template_note>

  <security_and_access_control>
    <user_roles>
      <role name="guest">
        <permissions>
          - Can access public pages (login, register, forgot password, 404)
          - Cannot access dashboard or private pages
          - Redirected to login when attempting to access private routes
        </permissions>
        <protected_routes>
          - /dashboard/* (redirects to /login)
        </protected_routes>
      </role>
      <role name="authenticated_user">
        <permissions>
          - Can access all dashboard pages
          - Can view and interact with all components
          - Can toggle dark/light mode
          - Can update profile settings
        </permissions>
        <protected_routes>
          - /login, /register (redirects to /dashboard if already logged in)
        </protected_routes>
      </role>
    </user_roles>
    <authentication>
      <method>Simulated email/password login with mock credentials + social login buttons (UI only)</method>
      <session_timeout>Simulated via localStorage/context</session_timeout>
      <password_requirements>Visual validation only (min 8 chars, shown in UI)</password_requirements>
    </authentication>
    <sensitive_operations>
      - Logout confirmation (optional)
    </sensitive_operations>
  </security_and_access_control>

  <mock_data_strategy>
    <description>
      All data in this template is mocked using static TypeScript files with typed mock data.
      No real API calls are made. Data is managed via React Context and local state.
      Mock data files are organized in a /data or /mocks directory.
    </description>
    <mock_data_files>
      - users.ts — Array of mock user objects (20-30 entries)
      - products.ts — Array of mock product objects (30-50 entries)
      - orders.ts — Array of mock order objects (20-40 entries)
      - messages.ts — Array of mock message/inbox objects (15-25 entries)
      - analytics.ts — Mock analytics data for charts
      - transactions.ts — Mock recent transactions
      - events.ts — Mock calendar events
      - kanban.ts — Mock kanban board data (columns and cards)
      - notifications.ts — Mock notification items
